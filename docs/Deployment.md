# 1️⃣ Check which branch you're on (usually "main" or "master")
git branch

# 2️⃣ See what files changed
git status

Verify the remote repository:
git remote -v 

# 3️⃣ Stage all modified files
git add .

# 4️⃣ Commit with a clear message
git commit -m "Fix favicon path for Vercel deployment"

# 5️⃣ Push to GitHub
git push origin main
